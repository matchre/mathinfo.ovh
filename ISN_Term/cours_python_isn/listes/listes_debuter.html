Le module random

Pour tester le bon comportement de certains programmes, il peut être utile de générer des nombres pseudo-aléatoires. En Python, le module random remplit ce rôle. Parmi les différentes fonctions qu'il contient, il faut particulièrement bien connaître :

  • randint(a, b) qui renvoie un entier compris entre les entiers a et b (inclus).
  • shuffle(liste) qui «mélange sur place» les éléments de la liste liste.

Apprendre à programmer, c'est aussi savoir reconstituer des fonctions déjà programmées par ailleurs. Une partie des exercices de cette page sont à réaliser dans cette optique.

from random import *

##----- Définition des Fonctions -----##
def liste_alea(n=10, p=100):
    """ Entrees : n et p sont deux entiers naturels non nuls
        Sorties : Renvoie une liste de n entiers aleatoires compris entre 1 et p"""
    resultat = []						# Liste vide
    for i in range(n):
        resultat.append(randint(1, p))
    return resultat


def inserer(liste, position, element):
    """ Entrees : une liste, position est un entier, element est non vide
                  position doit etre compris entre 0 (insertion en début de liste)
                  et len(liste) (insertion en fin de liste).
        Sorties : Renvoie la liste dans laquelle element est insere a l'indice position."""
    longueur = len(liste)
    if position <= 0:
        position = 0
    elif position >= longueur:
        position = longueur

    resultat = [0]*(longueur +1)			# Liste de 0 avec un élément de plus
    for i in range(position):
        resultat[i] = liste[i]				# Les premiers éléments sont recopiés
    resultat[position] = element
    for i in range(position, longueur):
        resultat[i+1] = liste[i]			# Les derniers élements sont décalés de 1

    return resultat
	

##----- Programme principal -----##
test = liste_alea(10, 6)
print(test)
print('L\'insertion dans la liste donne {}'.format(inserer(test, 4, -1)))

from random import *

##----- Définition des Fonctions -----##
def liste_alea(n=10, p=100):
    """ Entrees : n et p sont deux entiers naturels non nuls
        Sorties : Renvoie une liste de n entiers aleatoires compris entre 1 et p"""
    resultat = []						# Liste vide
    for i in range(n):
        resultat.append(randint(1, p))
    return resultat


def renverse(liste):
    """ Entrees : une liste.
        Sorties : renvoie la liste «inversee» : le 1er element devient le dernier,
                  le 2eme devient avant-dernier..."""
    resultat = []
    for i in liste:
        resultat = [i]+resultat		# On ajoute l'element en premiere position
    return resultat

##----- Importation des Modules -----##
from random import *

##----- Définition des Fonctions -----##
def liste_alea(n=10, p=100):
    """ Entrees : n et p sont deux entiers naturels non nuls
        Sorties : Renvoie une liste de n entiers aleatoires compris entre 1 et p"""
    resultat = []						# Liste vide
    for i in range(n):
        resultat.append(randint(1, p))
    return resultat


def maxi(liste):
    """ Entrees : une liste d'elements numeriques !
        Sorties : renvoie la valeur du plus grand element de la liste."""
    m = liste[0]			# Le maximum est le 1er element au depart
    for element in liste:
        if m < element :	# Si m est inferieur a un nouvel element
             m = element		# Alors m est remplace par cet element
    return m
	

##----- Programme principal -----##

print(test)

##----- Importation des Modules -----##
from random import *

##----- Définition des Fonctions -----##
def liste_alea(n=10, p=100):
    """ Entrees : n et p sont deux entiers naturels non nuls
        Sorties : Renvoie une liste de n entiers aleatoires compris entre 1 et p"""
    resultat = []						# Liste vide
    for i in range(n):
        resultat.append(randint(1, p))
    return resultat


def maxi_2(liste):
    """ Entrees : une liste d'elements numeriques !
        Sorties : renvoie une liste dont le 1er element est la valeur du plus grand element
                  de la liste d'origine et les elements suivants sont les indices ou le
                  maximum est present."""
    m = liste[0]			# Le maximum est le 1er element au depart
    resultat = []           # La liste des occurrences du maximum
	
    for i in range(len(liste)-1):	# On parcourt les indices
        if m < liste[i]:        	# Si m est inferieur a un nouvel element
            m = liste[i]        	# Alors m est remplace par cet element
            resultat = [i]          # et on re-initialise la liste des occurrences

        elif m == liste[i]:     	# Si le nouvel element est égal a m
            resultat.append(i)      # Son indice est ajoute a la liste

    resultat.insert(0, m)           # On place m au début de la liste a retourner
    return resultat

print('La liste est {}'.format(test))

maximum = maxi_2(test)
print('Le maximum de la liste est {}'.format(maximum.pop(0)))
print('Il se trouve aux indices {}'.format(maximum))

##----- Importation des Modules -----##
from random import *

##----- Définition des Fonctions -----##
def liste_alea(n=10, p=100):
    """ Entrees : n et p sont deux entiers naturels non nuls
        Sorties : Renvoie une liste de n entiers aleatoires compris entre 1 et p"""
    resultat = []						# Liste vide
    for i in range(n):
        resultat.append(randint(1, p))
    return resultat


def indices(element, liste):
    """ Entrees : element est une valeur, liste est une liste quelconque
        Sorties : renvoie la liste (eventuellement vide) des indices auxquels
                  se trouve element dans liste."""
    resultat = []           		# La liste des occurrences d'element
	
    for i in range(len(liste)-1):	# On parcourt les indices
        if element == liste[i]:     # Si l'element actuel est element
            resultat.append(i)    	# on ajoute l'indice dans le resultat

    return resultat


##----- Programme principal -----##
n = int(input('De quelle longueur doit etre la liste ? '))
test = liste_alea(n, 100)
print('La liste est {}'.format(test))

maximum = max(test)
print('Le maximum de la liste est {}'.format(maximum))

positions = indices(maximum, test)
print('Il se trouve aux indices {}'.format(positions))
